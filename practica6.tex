\section{Practica 6: Contador}

\subsection{Introducción}
\noindent
En esta práctica se programó el PIC16F628A con el objetivo de que sea un contador desde el número 1 al 99 usando 2 decodificadores uno para el puerto A y el otro para el puerto B.

\subsection{Materiales}
\noindent
Para la realización de esta práctica utilizamos los siguientes materiales y herramientas:
\begin{itemize}
	\item Microcontrolador PIC16F628A
	\item Display de 7 segmentos
	\item Resistencias
	\item Cristal de 4MHz
	\item Protoboard
	\item Cables de conexión
	\item 2 Decodificadores
	\item Fuente de alimentación
\end{itemize}

\subsection{Software}
\begin{itemize}
	\item Proteus 8.0
	\item MPLAB X IDE
	\item PICkit 3 Programmer App and Scripting Tool v3.10
\end{itemize}

\subsection{Codigo en C}
\begin{verbatim}
/*
 * File:   CONTADOR.c
 * Author: jesus
 *
 * Created on 17 de marzo de 2026, 10:51 a.m.
 */
// PIC16F628A Configuration Bit Settings

// 'C' source line config statements

// CONFIG
#pragma config FOSC = XT        // Oscillator Selection bits 
(XT oscillator: Crystal/resonator on RA6/OSC2/CLKOUT and RA7/OSC1/CLKIN)
#pragma config WDTE = OFF        // Watchdog Timer Enable bit (WDT enabled)
#pragma config PWRTE = ON       // Power-up Timer Enable bit (PWRT enabled)
#pragma config MCLRE = ON       // RA5/MCLR/VPP Pin Function Select bit 
(RA5/MCLR/VPP pin function is MCLR)
#pragma config BOREN = ON       // Brown-out Detect Enable bit (BOD enabled)
#pragma config LVP = OFF        // Low-Voltage Programming Enable bit 
(RB4/PGM pin has digital I/O function, HV on MCLR must be used for programming)
#pragma config CPD = OFF        // Data EE Memory Code Protection bit 
(Data memory code protection off)
#pragma config CP = OFF         // Flash Program Memory Code Protection bit (Code protection off)

// #pragma config statements should precede project file includes.
// Use project enums instead of #define for ON and OFF.

#include <xc.h>
#define _XTAL_FREQ 4000000
//para anodo
unsigned char numerosA[10] = {0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09};
unsigned char numerosB[10] = {0x00, 0x10, 0x20, 0x30, 0x40, 0x50, 0x60, 0x70, 0x80, 0x90};
//unsigned char numeros[16] = {0x3F, 0x06, 0x5B, 0x4F, 0x66, 0x6D, 0x7D, 0x07, 0x7F, 0x6F};

void main(void) {
    TRISA = 0X00;
    TRISB = 0X00;
    
    for(int i = 0; i < 10; i++){
        PORTA = numerosA[i];//decenas
        PORTB = 0x00;
        for(int j = 0; j < 10; j++){
            PORTB = numerosB[j];//unidades
            __delay_ms(100);
        }
    }
    return;
}
\end{verbatim}
